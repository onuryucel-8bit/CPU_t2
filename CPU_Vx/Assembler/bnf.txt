<program> ::= <opcode> (<adres> | <register>) (<register> | <sayi> | <adres>)*
<opcode> ::= "LOAD" | "STR" | "ADD"
<adres> ::= "@" <sayi>
<hex> ::= 0x<sayi>
<register> ::= "R" <sayi>
<sayi> ::= [0-9]
